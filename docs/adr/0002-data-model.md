# ADR 0002: Data Model for Monsoon Preparedness

## Status

Accepted

## Context

We need a data model that supports: personalized family preparedness plans, real-time weather alerts with localization, checklists per phase, SOS requests, community reports, risk scoring, and offline caching — all for a 1-district hackathon prototype with Supabase/PostgreSQL.

## Decision

Use the following PostgreSQL schema (Supabase-managed):

### Core Tables

```sql
-- Extends Supabase auth.users
CREATE TABLE profiles (
  id UUID PRIMARY KEY REFERENCES auth.users(id),
  phone TEXT UNIQUE NOT NULL,
  language TEXT DEFAULT 'hi',
  dialect TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE families (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  primary_member UUID REFERENCES profiles(id),
  address TEXT,
  lat DOUBLE PRECISION,
  lng DOUBLE PRECISION,
  district TEXT, -- 'Mumbai Suburban'
  ward TEXT,     -- 'Andheri West'
  hazard_zones TEXT[], -- ['flood-prone', 'waterlogging']
  housing_type TEXT, -- 'pucca', 'semi-pucca', 'kutcha', 'high-rise'
  vulnerability_score DECIMAL(3,2), -- 0.00-1.00 from indices
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE family_members (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  family_id UUID REFERENCES families(id) ON DELETE CASCADE,
  profile_id UUID REFERENCES profiles(id),
  relation TEXT, -- 'self', 'spouse', 'child', 'elderly_parent', 'pet'
  age INTEGER,
  medical_conditions TEXT[], -- ['diabetes', 'hypertension', 'mobility_impairment']
  is_vulnerable BOOLEAN GENERATED ALWAYS AS (
    age < 5 OR age > 65 OR array_length(medical_conditions, 1) > 0
  ) STORED
);

-- One plan per family per phase
CREATE TABLE preparedness_plans (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  family_id UUID REFERENCES families(id) ON DELETE CASCADE,
  phase TEXT CHECK (phase IN ('pre_monsoon', 'active_monsoon', 'post_monsoon')),
  generated_by TEXT DEFAULT 'genai', -- 'genai', 'manual'
  genai_prompt_hash TEXT, -- for reproducibility
  risk_score_at_generation DECIMAL(4,2),
  status TEXT DEFAULT 'active', -- 'active', 'completed', 'archived'
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(family_id, phase)
);

CREATE TABLE checklist_items (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  plan_id UUID REFERENCES preparedness_plans(id) ON DELETE CASCADE,
  item_text TEXT NOT NULL,
  item_text_localized JSONB, -- {"hi": "...", "mr": "...", "en": "..."}
  priority TEXT CHECK (priority IN ('critical', 'high', 'medium', 'low')),
  category TEXT, -- 'home_prep', 'documents', 'emergency_kit', 'evacuation', 'health', 'communication'
  phase_applicability TEXT[], -- which phases this applies to
  is_completed BOOLEAN DEFAULT FALSE,
  completed_at TIMESTAMPTZ,
  sort_order INTEGER DEFAULT 0
);

-- Mock IMD/CWC alerts for hackathon
CREATE TABLE weather_alerts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  source TEXT CHECK (source IN ('IMD', 'CWC', 'ULB', 'MOCK')),
  alert_type TEXT, -- 'heavy_rainfall', 'cyclone', 'flood', 'landslide', 'thunderstorm'
  severity TEXT CHECK (severity IN ('watch', 'alert', 'warning')),
  districts TEXT[], -- affected districts
  title TEXT,
  description TEXT,
  instruction TEXT, -- official recommended actions
  valid_from TIMESTAMPTZ,
  valid_until TIMESTAMPTZ,
  metadata JSONB, -- raw API payload
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Per-family localized alerts (generated by GenAI)
CREATE TABLE localized_alerts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  weather_alert_id UUID REFERENCES weather_alerts(id) ON DELETE CASCADE,
  family_id UUID REFERENCES families(id) ON DELETE CASCADE,
  language TEXT,
  title_localized TEXT,
  description_localized TEXT,
  action_items TEXT[], -- personalized actionable steps
  risk_score DECIMAL(4,2), -- 0-100 for this family
  generated_at TIMESTAMPTZ DEFAULT NOW(),
  read_at TIMESTAMPTZ
);

CREATE TABLE shelters (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT,
  address TEXT,
  lat DOUBLE PRECISION,
  lng DOUBLE PRECISION,
  district TEXT,
  ward TEXT,
  capacity INTEGER,
  current_occupancy INTEGER DEFAULT 0,
  facilities JSONB, -- {"water": true, "toilets": true, "medical": false, "food": false}
  managing_authority TEXT,
  is_active BOOLEAN DEFAULT TRUE
);

CREATE TABLE evacuation_routes (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  from_lat DOUBLE PRECISION,
  from_lng DOUBLE PRECISION,
  to_shelter_id UUID REFERENCES shelters(id),
  waypoints JSONB, -- [{"lat": ..., "lng": ..., "instruction": "..."}]
  distance_km DECIMAL(6,2),
  duration_min INTEGER,
  road_condition TEXT, -- 'good', 'waterlogged', 'blocked', 'unknown'
  suitable_for TEXT CHECK (suitable_for IN ('vehicle', 'pedestrian', 'both')),
  last_updated TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE sos_requests (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  profile_id UUID REFERENCES profiles(id),
  family_id UUID REFERENCES families(id),
  lat DOUBLE PRECISION,
  lng DOUBLE PRECISION,
  emergency_type TEXT CHECK (emergency_type IN ('medical', 'trapped', 'evacuation', 'other')),
  description TEXT,
  status TEXT CHECK (status IN ('active', 'responded', 'resolved', 'false_alarm')) DEFAULT 'active',
  created_at TIMESTAMPTZ DEFAULT NOW(),
  resolved_at TIMESTAMPTZ
);

CREATE TABLE community_reports (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  profile_id UUID REFERENCES profiles(id),
  lat DOUBLE PRECISION,
  lng DOUBLE PRECISION,
  report_type TEXT CHECK (report_type IN ('waterlogging', 'road_closure', 'landslide', 'shelter_status', 'power_outage')),
  severity TEXT CHECK (severity IN ('low', 'medium', 'high', 'critical')),
  description TEXT,
  photo_url TEXT,
  verification_status TEXT CHECK (verification_status IN ('unverified', 'verified', 'disputed')) DEFAULT 'unverified',
  upvotes INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Offline cache manifest per user device
CREATE TABLE offline_cache_manifests (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  profile_id UUID REFERENCES profiles(id),
  device_id TEXT, -- browser fingerprint
  checklist_items JSONB, -- cached checklist for current phase
  evacuation_routes JSONB, -- cached routes to nearest shelters
  shelters JSONB, -- cached shelter list
  emergency_contacts JSONB,
  last_synced TIMESTAMPTZ DEFAULT NOW()
);

-- Risk score history for trend visualization
CREATE TABLE risk_score_history (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  family_id UUID REFERENCES families(id) ON DELETE CASCADE,
  score DECIMAL(4,2),
  factors JSONB, -- {"weather": 0.7, "vulnerability": 0.4, "preparedness": 0.8}
  recorded_at TIMESTAMPTZ DEFAULT NOW()
);
```

### Row Level Security (RLS)

```sql
-- Families: members can read/update their family
ALTER TABLE families ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Family members can view" ON families
  FOR SELECT USING (id IN (SELECT family_id FROM family_members WHERE profile_id = auth.uid()));
CREATE POLICY "Primary member can update" ON families
  FOR UPDATE USING (primary_member = auth.uid());

-- Plans: family members can read
ALTER TABLE preparedness_plans ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Family members can view plans" ON preparedness_plans
  FOR SELECT USING (family_id IN (SELECT family_id FROM family_members WHERE profile_id = auth.uid()));

-- Checklist items: family members can read/update completion
ALTER TABLE checklist_items ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Family members can manage checklist" ON checklist_items
  FOR ALL USING (plan_id IN (
    SELECT id FROM preparedness_plans WHERE family_id IN (
      SELECT family_id FROM family_members WHERE profile_id = auth.uid()
    )
  ));

-- Localized alerts: family members can read
ALTER TABLE localized_alerts ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Family members can view alerts" ON localized_alerts
  FOR SELECT USING (family_id IN (SELECT family_id FROM family_members WHERE profile_id = auth.uid()));

-- SOS: creator can read/update, family members can read
ALTER TABLE sos_requests ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Creator can manage SOS" ON sos_requests
  FOR ALL USING (profile_id = auth.uid());
CREATE POLICY "Family can view SOS" ON sos_requests
  FOR SELECT USING (family_id IN (SELECT family_id FROM family_members WHERE profile_id = auth.uid()));
```

### Indexes for Performance

```sql
CREATE INDEX idx_weather_alerts_districts_valid ON weather_alerts USING GIN (districts) WHERE valid_until > NOW();
CREATE INDEX idx_localized_alerts_family_unread ON localized_alerts (family_id) WHERE read_at IS NULL;
CREATE INDEX idx_sos_requests_active ON sos_requests (family_id, status) WHERE status = 'active';
CREATE INDEX idx_community_reports_location ON community_reports USING GIST (ll_to_earth(lat, lng));
CREATE INDEX idx_evacuation_routes_shelter ON evacuation_routes (to_shelter_id);
```

## Consequences

### Positive

- Single database, no polyglot persistence
- RLS enforces data isolation at DB level (no app bugs leak data)
- JSONB for flexible localized content and GenAI metadata
- Supabase handles auth, realtime, edge functions
- Schema matches domain model in CONTEXT.md

### Negative

- JSONB queries less performant than normalized (acceptable for pilot scale)
- No separate read replica (Supabase free tier)
- Migrations manual for hackathon (acceptable)

### Risks

- Family-member circular FK (profiles → families → family_members → profiles) — resolved by making `families.primary_member` nullable on insert, updated after member creation
- Offline cache manifest grows — TTL cleanup job needed post-hackathon
